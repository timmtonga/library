  def save_book
        #this method saves new books to the database
        @book = Book.new (params[:book])

        if @book.save
                flash[:notice] = "Book details successfully added"
                redirect_to :controller => "book", :action => "add_book"
        else
                flash[:notice] = "Book details could not be saved"
                redirect_to :controller => "book", :action => "add_book"
        end

  end
